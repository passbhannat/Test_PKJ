CREATE PROC PROC_COPY_FROM_ARC
@CardCode varchar(50)
AS
SELECT 'N' "Chk",T0."DocNum", T0."U_ContNo" "Contract No",
T1."U_ItemCode" "ItemCode",T1."U_ItemName" "ItemName",
T1."U_Qty" "Quantity",T1."U_Price" "Price" 
FROM [@ARC] T0
INNER JOIN [@ARC1] T1 ON T0.DocEntry = T1.DocEntry
WHERE T0.U_CardCode = @CardCode
AND U_ItemCode IS NOT NULL

Sales Order SP Name: PROC_SO_DETAILS
Invoice Details SP Name: PROC_INVOICE_DETAILS
